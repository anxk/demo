# 字典树

字典树又叫前缀树，建树时间效率为log(NL)，查询时间效率为log(L)。

## 常见问题

- 字典树不仅可以用来处理字符串，也可以用来处理二进制串，每个节点可以存储单个字符也可以存储一个字符串
- 字符串查询和遍历

## 数组实现

```go
package main

// 字典树
type Trie struct {
    children [26]*Trie
    isEnd    bool
}

// 初始化一个字典树
func NewTrie() Trie {
    return Trie{}
}

// Insert 从根节点开始按输入word的字符依次向下延伸，若无代表次字符的结点，创建结点
func (t *Trie) Insert(word string) {
    root := t
    for _, ch := range word {
        ch -= 'a'
        if root.children[ch] == nil {
            root.children[ch] = &Trie{}
        }
        root = root.children[ch]
    }
    root.isEnd = true
}

// SearchPrefix 查找字串是否在前缀树上，找到则返回要查找的字串，否则返回null
func (t *Trie) SearchPrefix(prefix string) *Trie {
    root := t
    for _, ch := range prefix {
        ch -= 'a'
        if root.children[ch] == nil {
            return nil
        }
        root = root.children[ch]
    }
    // 这里返回的是最后一个字符代表的节点
    return root
}

// Search 向下寻找单词word，若找到，则末尾字符处判断是否为单词。
// 因此，Trie类还需维护一个isEnd属性，用以标记到该节点为止的前缀是否为单词
func (t *Trie) Search(word string) bool {
    root := t.SearchPrefix(word)
    return root != nil && root.isEnd
}

// StartsWith 向下寻找前缀prefix，某一字符找不到则立即返回false，否则返回true
func (t *Trie) StartsWith(prefix string) bool {
    return t.SearchPrefix(prefix) != nil
}
```

## 哈希表实现

```go
package main

// MAP实现方式
type Trie struct {
	children map[string]*Trie
	isEnd    bool
}

func NewTrie() *Trie {
	return &Trie{
		children: make(map[string]*Trie),
	}
}

func (t *Trie) Insert(word string) {
	root := t
	for _, ch := range word {
		if root.children[string(ch)] == nil {
			root.children[string(ch)] = &Trie{children: make(map[string]*Trie)}
		}
		root = root.children[string(ch)]
	}
	root.isEnd = true
}

func (t *Trie) SearchPrefix(prefix string) *Trie {
	root := t
	for _, ch := range prefix {
		if root.children[string(ch)] == nil {
			return nil
		}
		root = root.children[string(ch)]
	}
	return root
}

func (t *Trie) Search(word string) bool {
	root := t.SearchPrefix(word)
	return root != nil && root.isEnd
}

func (t *Trie) StartsWith(prefix string) bool {
	return t.SearchPrefix(prefix) != nil
}
```
