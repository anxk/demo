## LIS 最长递增子序列

```go
// 求 nums 种最长递增子序列
dp := make([]int, len(nums))
for i := 0; i < len(nums); i++ {
    for j := 0; j < i; j++ {
    if nums[i] > nums[j] {
            dp[i] = max(dp[i], dp[j]+1)
        }
    }
}
```

## LCS 最长公共子序列

```go
// 求s1和s2的最长公共字符串
dp := make([][]int, len(s1)+1)
for i := range dp {
    dp[i] = make([]int, len(s2)+1)
}
// 初始化
// pass
for i := 1; i < len(dp); i++ {
    for j := 1; j < len(dp[0]); j++ {
        if s1[i-1] == s2[j-1] {
            dp[i][j] = dp[i-1][j-1] + 1
        } else {
            dp[i][j] = max(dp[i-1][j], dp[i][j-1])
        }
    }
}
```

## 01背包问题

```go
// 背包容量V，物品种类N，物品体积 v = [1, 2, 3]，物品价值 m = [3, 4, 5]，求背包最大能装物品的价值总和M
// 二维数组实现
dp := make([][]int, N)
for i := 0; i < len(dp); i++ {
    dp[i] = make([]int, V+1)
}
for j := v[0]; j < len(dp[0]); j++ {
    dp[0][j] = m[0]
}
for i := 1; i < len(dp); i++ {
    for j := 1; j < len(dp[0]); j++ {
        if j >= v[i] {
            dp[i][j] = max(dp[i-1][j], dp[i-1][j-v[i]]+m[i])
        } else {
            dp[i][j] = dp[i-1][j]
        }
    }
}
// M = dp[N-1][V]

// 滚动数组实现
dp = make([]int, V+1)
for i := 0; i < N; i++ {
    for j := len(dp)-1; j >= v[i]; j-- {
        dp[j] = max(dp[j], dp[j-v[i]]+m[i])
    }
}
```

## 完全背包

```go
// 背包容量V，物品种类N，每种物品有无数个，物品体积 v = [1, 2, 3]，物品价值 m = [3, 4, 5]，求背包最大能装物品的价值总和M
// 二维数组实现
dp := make([][]int, N)
for i := 0; i < len(dp); i++ {
    dp[i] = make([]int, V+1)
}
for j := v[0]; j < len(dp[0]); j++ {
    dp[0][j] = m[0]
}
for i := 1; i < len(dp); i++ {
    for j := 1; j < len(dp[0]); j++ {
        if j >= v[i] {
                        // 注意和01背包的差别
            dp[i][j] = max(dp[i-1][j], dp[i][j-v[i]]+m[i])
        } else {
            dp[i][j] = dp[i-1][j]
        }
    }
}
// 滚动数组实现，和01背包遍历背包顺序相反
dp = make([]int, V+1)
for i := 0; i < N; i++ {
    for j := v[i]; j < len(dp); j++ {
        dp[j] = max(dp[j], dp[j-v[i]]+m[i])
    }
}
```
